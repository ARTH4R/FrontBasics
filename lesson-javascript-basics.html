JavaScript พื้นฐาน
เริ่มต้นเรียนการเขียนคำสั่งและโปรแกรมเบื้องต้น
JavaScript คือภาษาโปรแกรมที่ทำให้เว็บไซต์ มีชีวิตชีวา และมีปฏิสัมพันธ์กับผู้ใช้ ทำงานบน browser (Client-side) และ server (Node.js) เป็น 1 ใน 3 ภาษาหลักของ web development: HTML, CSS, JavaScript JavaScript ทำให้เราสามารถ จัดการข้อมูล คำนวณ ตอบสนองต่อผู้ใช้ และสร้าง web applications ที่ซับซ้อนได้

ตัวแปร (Variables) - เก็บข้อมูล

ตัวแปร คือ "กล่อง" สำหรับเก็บข้อมูล มี 3 วิธีประกาศ: var, let, const let - เปลี่ยนค่าได้ (แนะนำใช้) const - เปลี่ยนค่าไม่ได้ (ค่าคงที่) var - วิธีเก่า ไม่แนะนำในโค้ดใหม่
ประเภท รูปแบบ เปลี่ยนค่าได้ Scope ใช้เมื่อ
let let name = "John"; ✅ ได้ Block scope ค่าที่ต้องเปลี่ยนได้
const const PI = 3.14; ❌ ไม่ได้ Block scope ค่าคงที่ ไม่เปลี่ยน
var var age = 25; ✅ ได้ Function scope โค้ดเก่า (หลีกเลี่ยง)
ตัวอย่างการใช้งาน: // let - เปลี่ยนค่าได้
let score = 0;
score = 100; // ✅ ได้
score = 200; // ✅ ได้

// const - เปลี่ยนค่าไม่ได้
const MAX_USERS = 100;
MAX_USERS = 200; // ❌ Error!

// const กับ object/array - แก้ properties ได้
const user = { name: "John" };
user.name = "Jane"; // ✅ ได้
user = {}; // ❌ Error! (ห้ามเปลี่ยนตัวแปร)

// การตั้งชื่อตัวแปร
let userName = "John"; // ✅ camelCase (แนะนำ)
let user_name = "John"; // ✅ snake_case
let UserName = "John"; // ✅ PascalCase (ใช้กับ class)
let 1name = "John"; // ❌ Error! ขึ้นต้นด้วยตัวเลขไม่ได้ หลักการตั้งชื่อที่ดี: - ใช้ชื่อที่มีความหมาย: userAge ดีกว่า x - camelCase สำหรับตัวแปรและฟังก์ชัน - UPPER_CASE สำหรับค่าคงที่: const MAX_SIZE = 100; - ห้ามใช้คำสงวน: let, const, function, if, else เป็นต้น

ชนิดข้อมูล (Data Types)

JavaScript มี 2 กลุ่มชนิดข้อมูล: Primitive และ Reference
ชนิดข้อมูล คำอธิบาย ตัวอย่าง typeof
String ข้อความ "Hello", 'World', `Template` "string"
Number ตัวเลข (จำนวนเต็มและทศนิยม) 42, 3.14, -7, Infinity, NaN "number"
Boolean ค่าความจริง true, false "boolean"
Undefined ไม่ได้กำหนดค่า undefined "undefined"
Null ค่าว่าง (ตั้งใจให้ว่าง) null "object" (bug ในภาษา)
Symbol ค่าที่ไม่ซ้ำ (ES6) Symbol("id") "symbol"
BigInt เลขจำนวนเต็มขนาดใหญ่ 9007199254740991n "bigint"
Object กลุ่มข้อมูล {name: "John", age: 30} "object"
Array ลิสต์ข้อมูล [1, 2, 3, 4, 5] "object"
Function ฟังก์ชัน function() { } "function"
ตัวอย่างการใช้งาน: // String - ข้อความ
let firstName = "John";
let lastName = 'Doe';
let fullName = `${firstName} ${lastName}`; // Template literal

// Number - ตัวเลข
let age = 25;
let price = 99.99;
let negative = -10;
let result = 10 / 0; // Infinity
let notANumber = "abc" / 2; // NaN

// Boolean - จริง/เท็จ
let isActive = true;
let hasPermission = false;
let isAdult = age >= 18; // true

// Undefined vs Null
let x; // undefined (ยังไม่ได้กำหนดค่า)
let y = null; // null (ตั้งใจให้ค่าว่าง)

// Object
let person = {
  name: "John",
  age: 30,
  city: "Bangkok"
};

// Array
let colors = ["red", "green", "blue"];
let numbers = [1, 2, 3, 4, 5];
let mixed = [1, "text", true, null]; // ได้หลายชนิด

// ตรวจสอบชนิดข้อมูล
console.log(typeof "Hello"); // "string"
console.log(typeof 42); // "number"
console.log(typeof true); // "boolean"
console.log(typeof undefined); // "undefined"
console.log(typeof null); // "object" (bug!)
console.log(typeof {}); // "object"
console.log(typeof []); // "object"
console.log(Array.isArray([])); // true (ตรวจสอบ array)

ตัวดำเนินการ (Operators)

1. ตัวดำเนินการทางคณิตศาสตร์ (Arithmetic Operators)
ตัวดำเนินการ ชื่อ ตัวอย่าง ผลลัพธ์
+ บวก 10 + 5 15
- ลบ 10 - 5 5
* คูณ 10 * 5 50
/ หาร 10 / 5 2
% หารเอาเศษ (modulus) 10 % 3 1
** ยกกำลัง 2 ** 3 8 (2³)
++ เพิ่มค่า 1 let x = 5; x++; 6
-- ลดค่า 1 let x = 5; x--; 4
2. ตัวดำเนินการเปรียบเทียบ (Comparison Operators)
ตัวดำเนินการ ชื่อ ตัวอย่าง ผลลัพธ์
== เท่ากับ (ค่า) 5 == "5" true
=== เท่ากับ (ค่าและชนิด) 5 === "5" false
!= ไม่เท่ากับ (ค่า) 5 != "5" false
!== ไม่เท่ากับ (ค่าและชนิด) 5 !== "5" true
> มากกว่า 10 > 5 true
< น้อยกว่า 10 < 5 false
>= มากกว่าหรือเท่ากับ 10 >= 10 true
<= น้อยกว่าหรือเท่ากับ 10 <= 5 false
3. ตัวดำเนินการตรรกะ (Logical Operators)
ตัวดำเนินการ ชื่อ ตัวอย่าง ผลลัพธ์
&& และ (AND) true && false false (ต้องจริงทั้งหมด)
|| หรือ (OR) true || false true (จริงอย่างน้อย 1 ตัว)
! ไม่ (NOT) !true false (กลับค่า)
ตัวอย่างการใช้งาน: // Arithmetic
let sum = 10 + 5; // 15
let remainder = 10 % 3; // 1
let power = 2 ** 3; // 8

let counter = 0;
counter++; // 1
counter += 5; // 6

// Comparison - ใช้ === แทน ==
console.log(5 == "5"); // true (แนะนำหลีกเลี่ยง)
console.log(5 === "5"); // false (✅ แนะนำ)
console.log(5 === 5); // true

// Logical
let age = 25;
let hasLicense = true;

// AND - ต้องจริงทั้งหมด
if (age >= 18 && hasLicense) {
  console.log("สามารถขับรถได้");
}

// OR - จริงอย่างน้อย 1 ตัว
if (age < 18 || !hasLicense) {
  console.log("ไม่สามารถขับรถได้");
}

// NOT - กลับค่า
let isLoggedIn = false;
if (!isLoggedIn) {
  console.log("กรุณาเข้าสู่ระบบ");
}

เงื่อนไข (Conditionals) - if/else

if/else ใช้ตัดสินใจว่าจะทำอะไรตามเงื่อนไข รูปแบบต่างๆ: // if เดียว
if (condition) {
  // ทำถ้าเงื่อนไขเป็นจริง
}

// if-else
if (condition) {
  // ทำถ้าเงื่อนไขเป็นจริง
} else {
  // ทำถ้าเงื่อนไขเป็นเท็จ
}

// if-else if-else
if (condition1) {
  // ทำถ้าเงื่อนไข 1 เป็นจริง
} else if (condition2) {
  // ทำถ้าเงื่อนไข 2 เป็นจริง
} else {
  // ทำถ้าทุกเงื่อนไขเป็นเท็จ
}

// Ternary operator (if-else แบบสั้น)
let result = condition ? valueIfTrue : valueIfFalse; ตัวอย่างการใช้งาน: // ตรวจสอบอายุ
let age = 20;

if (age >= 18) {
  console.log("บรรลุนิติภาวะ");
} else {
  console.log("ยังไม่บรรลุนิติภาวะ");
}

// คะแนนเกรด
let score = 75;
let grade;

if (score >= 80) {
  grade = "A";
} else if (score >= 70) {
  grade = "B";
} else if (score >= 60) {
  grade = "C";
} else if (score >= 50) {
  grade = "D";
} else {
  grade = "F";
}

console.log(`เกรด: ${grade}`); // "เกรด: B"

// Ternary operator
let status = age >= 18 ? "ผู้ใหญ่" : "เด็ก";
let discount = isMember ? 0.2 : 0; // ลด 20% ถ้าเป็นสมาชิก

// Nested ternary (หลีกเลี่ยงถ้าซับซ้อน)
let message = score >= 50
  ? score >= 80 ? "ดีมาก" : "ผ่าน"
  : "ไม่ผ่าน";

Switch Statement - เลือกทางเดินหลายทาง

switch เหมาะสำหรับเปรียบเทียบค่าเดียวกับหลายๆ case switch (expression) {
  case value1:
    // ทำถ้า expression === value1
    break;
  case value2:
    // ทำถ้า expression === value2
    break;
  default:
    // ทำถ้าไม่ตรงกับ case ใดๆ
} ตัวอย่างการใช้งาน: // วันในสัปดาห์
let day = 3;
let dayName;

switch (day) {
  case 1:
    dayName = "จันทร์";
    break;
  case 2:
    dayName = "อังคาร";
    break;
  case 3:
    dayName = "พุธ";
    break;
  case 4:
    dayName = "พฤหัสบดี";
    break;
  case 5:
    dayName = "ศุกร์";
    break;
  case 6:
    dayName = "เสาร์";
    break;
  case 7:
    dayName = "อาทิตย์";
    break;
  default:
    dayName = "ไม่ถูกต้อง";
}

// รวม case (fall-through)
let month = 2;
let season;

switch (month) {
  case 12:
  case 1:
  case 2:
    season = "ฤดูหนาว";
    break;
  case 3:
  case 4:
  case 5:
    season = "ฤดูร้อน";
    break;
  case 6:
  case 7:
  case 8:
  case 9:
    season = "ฤดูฝน";
    break;
  default:
    season = "ไม่ทราบ";
}

การวนซ้ำ (Loops)

Loops ใช้ทำงานซ้ำๆ จนกว่าเงื่อนไขจะไม่เป็นจริง
ประเภท รูปแบบ ใช้เมื่อ ตัวอย่าง
for for (init; condition; increment) รู้จำนวนรอบ วนซ้ำ 10 รอบ
while while (condition) ไม่รู้จำนวนรอบ วนจนกว่าจะ input ถูก
do-while do { } while (condition) ต้องทำอย่างน้อย 1 รอบ แสดง menu อย่างน้อย 1 ครั้ง
for...of for (item of array) วนใน array/string วนทุกสมาชิกใน array
for...in for (key in object) วนใน object properties วนทุก key ใน object
ตัวอย่างการใช้งาน: // for loop - แบบคลาสสิก
for (let i = 0; i < 5; i++) {
  console.log(`รอบที่ ${i + 1}`);
}
// Output: รอบที่ 1, รอบที่ 2, ..., รอบที่ 5

// while loop
let count = 0;
while (count < 5) {
  console.log(`Count: ${count}`);
  count++;
}

// do-while loop - ทำก่อนอย่างน้อย 1 รอบ
let num = 10;
do {
  console.log(`Number: ${num}`);
  num++;
} while (num < 5); // แม้เงื่อนไขเป็นเท็จ ก็ทำ 1 รอบ

// for...of - วน array
const fruits = ["apple", "banana", "orange"];
for (const fruit of fruits) {
  console.log(fruit);
}
// Output: apple, banana, orange

// for...in - วน object
const person = { name: "John", age: 30, city: "Bangkok" };
for (const key in person) {
  console.log(`${key}: ${person[key]}`);
}
// Output: name: John, age: 30, city: Bangkok

// break - หยุด loop
for (let i = 0; i < 10; i++) {
  if (i === 5) {
    break; // หยุดเมื่อ i = 5
  }
  console.log(i);
}
// Output: 0, 1, 2, 3, 4

// continue - ข้าม iteration
for (let i = 0; i < 5; i++) {
  if (i === 2) {
    continue; // ข้าม i = 2
  }
  console.log(i);
}
// Output: 0, 1, 3, 4

ฟังก์ชัน (Functions) - บล็อกของโค้ดที่ใช้ซ้ำได้

Function คือกลุ่มคำสั่งที่ทำงานเฉพาะ สามารถเรียกใช้ซ้ำได้ รูปแบบการประกาศ: // 1. Function Declaration
function greet(name) {
  return `สวัสดี ${name}`;
}

// 2. Function Expression
const greet = function(name) {
  return `สวัสดี ${name}`;
};

// 3. Arrow Function (ES6)
const greet = (name) => {
  return `สวัสดี ${name}`;
};

// Arrow function แบบสั้น
const greet = name => `สวัสดี ${name}`;
รูปแบบ Hoisting this binding ใช้เมื่อ
Function Declaration ✅ ได้ Dynamic ฟังก์ชันหลัก, ต้องการ hoisting
Function Expression ❌ ไม่ได้ Dynamic เก็บฟังก์ชันในตัวแปร
Arrow Function ❌ ไม่ได้ Lexical (ไม่มี this) ฟังก์ชันสั้นๆ, callbacks
ตัวอย่างการใช้งาน: // ฟังก์ชันพื้นฐาน
function add(a, b) {
  return a + b;
}

let result = add(5, 3); // 8

// Parameters vs Arguments
function greet(name, age) { // name, age = parameters
  console.log(`สวัสดี ${name} อายุ ${age} ปี`);
}

greet("John", 25); // "John", 25 = arguments

// Default parameters
function multiply(a, b = 1) {
  return a * b;
}

console.log(multiply(5)); // 5 (ใช้ค่า default b = 1)
console.log(multiply(5, 2)); // 10

// Rest parameters - รับหลายค่า
function sum(...numbers) {
  return numbers.reduce((total, num) => total + num, 0);
}

console.log(sum(1, 2, 3, 4, 5)); // 15

// Arrow function ตัวอย่าง
const numbers = [1, 2, 3, 4, 5];

// แบบเต็ม
const doubled = numbers.map((num) => {
  return num * 2;
});

// แบบสั้น
const doubled = numbers.map(num => num * 2);

// Return object (ต้องใส่ วงเล็บ)
const createUser = (name, age) => ({ name, age });
console.log(createUser("John", 30)); // {name: "John", age: 30}

อาร์เรย์ (Arrays) - เก็บข้อมูลหลายตัว

Array คือตัวแปรที่เก็บข้อมูลหลายค่าในตัวเดียว การสร้างและใช้งาน: // สร้าง array
const fruits = ["apple", "banana", "orange"];
const numbers = [1, 2, 3, 4, 5];
const mixed = [1, "text", true, null, {name: "John"}];
const empty = [];

// เข้าถึงข้อมูล (index เริ่มที่ 0)
console.log(fruits[0]); // "apple"
console.log(fruits[1]); // "banana"
console.log(fruits[fruits.length - 1]); // "orange" (ตัวสุดท้าย)

// แก้ไขข้อมูล
fruits[1] = "mango";
console.log(fruits); // ["apple", "mango", "orange"]

// Properties และ Methods
console.log(fruits.length); // 3 (จำนวนสมาชิก) Array Methods ที่สำคัญ:
Method คำอธิบาย ตัวอย่าง ผลลัพธ์
push() เพิ่มท้าย arr.push(4) [1,2,3,4]
pop() ลบท้าย arr.pop() [1,2] (ลบ 3 ออก)
unshift() เพิ่มหน้า arr.unshift(0) [0,1,2,3]
shift() ลบหน้า arr.shift() [2,3] (ลบ 1 ออก)
splice() เพิ่ม/ลบตรงกลาง arr.splice(1, 1) [1,3] (ลบตำแหน่ง 1)
slice() คัดลอกส่วนหนึ่ง arr.slice(0, 2) [1,2] (ไม่แก้ของเดิม)
concat() รวม array arr1.concat(arr2) [1,2,3,4,5]
indexOf() หาตำแหน่ง arr.indexOf(2) 1 (ไม่เจอ = -1)
includes() ตรวจสอบมีหรือไม่ arr.includes(2) true
join() แปลงเป็น string arr.join(", ") "1, 2, 3"
reverse() กลับลำดับ arr.reverse() [3,2,1] (แก้ของเดิม)
sort() เรียงลำดับ arr.sort() [1,2,3] (แก้ของเดิม)
Modern Array Methods (ES6+): const numbers = [1, 2, 3, 4, 5];

// map - แปลงทุกตัว
const doubled = numbers.map(num => num * 2);
// [2, 4, 6, 8, 10]

// filter - กรองตามเงื่อนไข
const evenNumbers = numbers.filter(num => num % 2 === 0);
// [2, 4]

// find - หาตัวแรกที่ตรง
const found = numbers.find(num => num > 3);
// 4

// reduce - รวมเป็นค่าเดียว
const sum = numbers.reduce((total, num) => total + num, 0);
// 15

// forEach - วนทุกตัว
numbers.forEach(num => console.log(num));

// some - มีอย่างน้อย 1 ตัวที่ตรง
const hasEven = numbers.some(num => num % 2 === 0);
// true

// every - ทุกตัวต้องตรง
const allPositive = numbers.every(num => num > 0);
// true

ออบเจ็กต์ (Objects) - เก็บข้อมูลแบบ key-value

Object คือโครงสร้างข้อมูลที่เก็บคู่ของ key-value การสร้างและใช้งาน: // สร้าง object
const person = {
  name: "John",
  age: 30,
  city: "Bangkok",
  isStudent: false,
  hobbies: ["reading", "coding"]
};

// เข้าถึงข้อมูล
console.log(person.name); // "John" (dot notation)
console.log(person["age"]); // 30 (bracket notation)

// แก้ไขข้อมูล
person.age = 31;

1. คำนวณคะแนนเฉลี่ย function calculateAverage(scores) {
  if (scores.length === 0) return 0;
  
  const sum = scores.reduce((total, score) => total + score, 0);
  return sum / scores.length;
}

const myScores = [85, 90, 78, 92, 88];

  // ทำให้เป็นตัวพิมพ์เล็กและลบช่องว่าง
  const cleaned = str.toLowerCase().replace(/\s/g, '');
  const reversed = cleaned.split('').reverse().join('');
  return cleaned === reversed;
}

Best Practices - แนวทางปฏิบัติที่ดี

1. ตัวแปร - ใช้ const เป็นค่าเริ่มต้น เปลี่ยนเป็น let ถ้าต้องเปลี่ยนค่า - หลีกเลี่ยง var ในโค้ดใหม่ - ตั้งชื่อให้มีความหมาย: userAge ดีกว่า x - ใช้ camelCase: firstName, isActive 2. การเปรียบเทียบ // ❌ หลีกเลี่ยง

// ✅ แนะนำ

  if (age) {
    if (age >= 18) {
      return "ผู้ใหญ่";
    } else {
      return "เด็ก";
    }
  }
}

  if (!age) return null;
  if (age >= 18) return "ผู้ใหญ่";
  return "เด็ก";
} 4. Arrays และ Objects - ใช้ modern methods: map, filter, reduce - หลีกเลี่ยง mutation ของ array: ใช้ methods ที่ return array ใหม่ - ใช้ destructuring เพื่อความสะดวก - ใช้ spread operator แทน Object.assign() // ❌ Mutation
